rds_setting_flag_converter = {
	if = {
		limit = {
			is_scope_type = fleet
		}
		if = {
			limit = {
				owner = {
					has_country_flag = $RDS_SETTING_FLAG$
				}
			}
			set_fleet_flag = $RDS_SETTING_FLAG$
		}
		else = {
			fleet = {
				remove_fleet_flag = $RDS_SETTING_FLAG$
			}
		}
	}
	else_if = {
		limit = { 
			is_scope_type = ship
		}
		if = {
			limit = {
				fleet = {
					has_fleet_flag = $RDS_SETTING_FLAG$
				}
			}
			set_ship_flag = $RDS_SETTING_FLAG$
		}
		else = {
			remove_ship_flag = $RDS_SETTING_FLAG$
		}
	}
}

dynamic_flag_set = {
	if = {
		limit = {
			is_scope_type = country
		}
		set_country_flag = $DYNAMIC_FLAG$
	}
	else_if = {
		limit = { 
			is_scope_type = fleet
		}
		set_fleet_flag = $DYNAMIC_FLAG$
	}
	else_if = {
		limit = {
			is_scope_type = ship
		}
		set_ship_flag = $DYNAMIC_FLAG$
	}
}

dynamic_flag_remove = {
	if = {
		limit = {
			is_scope_type = country
		}
		remove_country_flag = $DYNAMIC_FLAG$
	}
	else_if = {
		limit = { 
			is_scope_type = fleet
		}
		remove_fleet_flag = $DYNAMIC_FLAG$
	}
	else_if = {
		limit = {
			is_scope_type = ship
		}
		remove_ship_flag = $DYNAMIC_FLAG$
	}
}

dynamic_event = {
	if = {
		limit = {
			is_scope_type = country
		}
		country_event = {
			id = $EVENT$
		}
	}
	else_if = {
		limit = { 
			is_scope_type = fleet
		}
		fleet_event = {
			id = $EVENT$
		}
	}
	else_if = {
		limit = { 
			is_scope_type = ship
		}
		ship_event = {
			id = $EVENT$
		}
	}
}
