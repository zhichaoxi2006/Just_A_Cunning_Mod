dynamic_flag_check = {
	if = {
		limit = {
			is_scope_type = country
		}
		has_country_flag = $DYNAMIC_FLAG$
	}
	else_if = {
		limit = { 
			is_scope_type = fleet
		}
		has_fleet_flag = $DYNAMIC_FLAG$
	}
	else_if = {
		limit = { 
			is_scope_type = ship
		}
		has_ship_flag = $DYNAMIC_FLAG$
	}
}

EQUIPPED_WITH_RDS = {
	OR = {
		has_component = REALITY_DISTORTION_SYSTEM 
		leader = {
			OR = {
				has_trait = leader_trait_rds
				has_leader_flag = EQUIPPED_WITH_RDS
			}
		}
	}
}

has_rds_flag = {
	fleet = {
		has_fleet_flag = $FLAG$
	}
}
